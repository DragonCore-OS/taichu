> **说明**  
> * 模型方程（基线）  
> \[
> \begin{aligned}
> \frac{dS}{dt}&=-\beta(t)\frac{SI}{N}+\gamma R-\delta S,\\[4pt]
> \frac{dI}{dt}&=\beta(t)\frac{SI}{N}-(\alpha+\mu)I,\\[4pt]
> \frac{dR}{dt}&=\alpha I-\gamma R,\\[4pt]
> \frac{dD}{dt}&=\mu I+\delta S,
> \end{aligned}
> \]  
> 其中  
> \(\displaystyle \beta(t)=0.5\bigl(1-e^{-0.1t}\bigr)\)，  
> \(\displaystyle \alpha=\frac1{14}\approx0.0714286\;{\rm d^{-1}}\)（康复率），

#!/usr/bin/env python
# -*- coding: utf-8 -*-

import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

# ------------------ 参数 ------------------
N = 1e7                     # 总人口
beta0 = 0.5                 # β(t) 的上限
k_beta = 0.1                # β(t) 上升速率
alpha = 1/14                # 康复率 (1/d)
mu    = 0.01                # 病死率 (1/d)
gamma = 1/90                # 免疫失效率 (1/d)
delta = 0.001               # 易感者自然减少率 (1/d)

# 疫苗接种参数（在第 50 天起生效）
vacc_start = 50.0           # 天
vacc_rate  = 0.05           # 每天转移比例 (1/d)

# 初始条件
I0 = 100.0
S0 = N - I0
R0 = 0.0
D0 = 0.0
y0 = [S0, I0, R0, D0]

# ------------------ β(t) ------------------
def beta(t):
    """时间依赖的传染率"""
    return beta0 * (1.0 - np.exp(-k_beta * t))

# ------------------ 右端函数 ------------------
def sird_rhs(t, y, vaccinate=False):
    S, I, R, D = y
    bet = beta(t)

    # 基线项
    dS = -bet * S * I / N + gamma * R - delta * S
    dI =  bet * S * I / N - (alpha + mu) * I
    dR =  alpha * I - gamma * R
    dD =  mu * I + delta * S

    # 疫苗接种项（仅在 t>=vacc_start 且 vaccinate=True 时加入）
    if vaccinate and t >= vacc_start:
        v = vacc_rate * S          # 每天转移到免疫者的人数
        dS -= v                  # S 减少
        dR += v                  # R 增加
        # D 不受直接影响（假设疫苗完全有效且不致死）

    return [dS, dI, dR, dD]

# ------------------ 求解函数 ------------------
def run_simulation(vaccinate=False, t_end=200):
    sol = solve_ivp(
        fun=lambda t, y: sird_rhs(t, y, vaccinate=vaccinate),
        t_span=(0, t_end),
        y0=y0,
        method='RK45',
        dense_output=True,
        max_step=0.5,          # 为了获得平滑曲线，步长不超过0.5天
        rtol=1e-8,
        atol=1e-10
    )
    return sol

# ------------------ 基线模拟（无疫苗） ------------------
sol_base = run_simulation(vaccinate=False)

# ------------------ 带疫苗模拟（第50天起） ------------------
sol_vac  = run_simulation(vaccinate=True)

# ------------------ 辅助函数：提取峰值 ------------------
def peak_info(sol):

    t = sol.t
    idx_max = np.argmax(I)
    return t[idx_max], I[idx_max]

t_peak_base, I_peak_base = peak_info(sol_base)
t_peak_vac , I_peak_vac  = peak_info(sol_vac)
